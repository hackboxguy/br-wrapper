loaddtb=ext2load mmc 0:2 ${devicetree_load_address} ${devicetree_image}
bootargs=earlyprintk console=ttyPS0,115200 root=/dev/mmcblk0p2 ro rootwait rootfs=ext2
bitstream_image=zynq-zed-top.bin
bitstream_type=load
fpga_config=fpga ${bitstream_type} 0 ${bitstream_load_address} ${filesize}
